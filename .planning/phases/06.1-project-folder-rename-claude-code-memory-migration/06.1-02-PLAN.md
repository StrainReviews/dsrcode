**Script header:**
```powershell
#Requires -Version 5.1
<#
.SYNOPSIS
    Phase 6.1 prerequisite checker — verifies no process blocks the directory rename.

.DESCRIPTION
    Validates that no process holds a handle on C:\Users\ktown\Projects\cc-discord-presence
    or runs from within the project directory. Checks:
      1. dsrcode/cc-discord-presence daemon (PID 55564 or by name)
      2. Claude Code processes (claude.exe, "Claude Code.exe")
      3. File handles via Sysinternals Handle.exe (optional)
      4. Running go build/test/dlv/gopls processes
      5. PowerShell/Git Bash shells with cwd in project dir

.PARAMETER DryRun
    Run checks but exit 0 regardless of results (for test runs).

.PARAMETER Verbose
    Emit detailed output for each check.

.EXAMPLE
    .\prereq-check.ps1
    .\prereq-check.ps1 -DryRun
    .\prereq-check.ps1 -Verbose

.NOTES
    Exit codes:
      0 = OK (safe to rename)
      1 = Warnings (user should investigate but not blocking)
      2 = BLOCKERS (rename will fail — fix before running handoff.ps1)
#>

[CmdletBinding()]
param(
    [switch]$DryRun
)

$ErrorActionPreference = 'Continue'
$ProjectPath = 'C:\Users\ktown\Projects\cc-discord-presence'
$DaemonPid = 55564

$blockers = New-Object System.Collections.Generic.List[string]
$warnings = New-Object System.Collections.Generic.List[string]

function Write-Check {
    param([string]$Name, [string]$Status, [string]$Detail)
    $color = switch ($Status) {
        'OK'    { 'Green' }
        'WARN'  { 'Yellow' }
        'BLOCK' { 'Red' }
        default { 'White' }
    }
    Write-Host "[$Status] $Name" -ForegroundColor $color
    if ($Detail) { Write-Host "       $Detail" -ForegroundColor Gray }
}

Write-Host "Phase 6.1 Prereq Check — $ProjectPath"
Write-Host "=========================================="

# Check 1: Daemon PID 55564
$daemon = Get-Process -Id $DaemonPid -ErrorAction SilentlyContinue
if ($daemon) {
    $blockers.Add("Daemon PID $DaemonPid running: $($daemon.ProcessName) ($($daemon.Path))")
    Write-Check -Name "Daemon PID $DaemonPid" -Status 'BLOCK' -Detail "$($daemon.ProcessName) at $($daemon.Path)"
} else {
    Write-Check -Name "Daemon PID $DaemonPid" -Status 'OK' -Detail 'Not running'
}

# Check 2: Daemon by name (in case PID changed)
$daemonByName = Get-Process -Name 'dsrcode','cc-discord-presence','cc-discord-presence-windows-amd64' -ErrorAction SilentlyContinue
if ($daemonByName) {
    foreach ($p in $daemonByName) {
        $blockers.Add("Daemon process by name: PID $($p.Id) $($p.ProcessName)")
    }
    Write-Check -Name "Daemon by name" -Status 'BLOCK' -Detail ($daemonByName | ForEach-Object { "PID $($_.Id)" }) -join ', '
} else {
    Write-Check -Name "Daemon by name" -Status 'OK' -Detail 'No dsrcode/cc-discord-presence processes'
}

# Check 3: Claude Code processes
$claudeProcs = Get-Process -Name 'claude','Claude Code','node' -ErrorAction SilentlyContinue |
               Where-Object { $_.Path -like '*claude*' -or $_.ProcessName -eq 'claude' }
if ($claudeProcs) {
    foreach ($p in $claudeProcs) {
        $warnings.Add("Claude Code process: PID $($p.Id) $($p.ProcessName)")
    }
    Write-Check -Name "Claude Code processes" -Status 'WARN' -Detail ("$($claudeProcs.Count) processes")
} else {
    Write-Check -Name "Claude Code processes" -Status 'OK' -Detail 'None running'
}

# Check 4: Processes with cwd in project (CIM fallback for Handle.exe)
try {
    $cwdProcs = Get-CimInstance -ClassName Win32_Process -ErrorAction Stop |
                Where-Object { $_.CommandLine -like "*cc-discord-presence*" -and $_.ProcessId -ne $PID }
    if ($cwdProcs) {
        foreach ($p in $cwdProcs) {
            $blockers.Add("Process with project path in CommandLine: PID $($p.ProcessId) $($p.Name)")
        }
        Write-Check -Name "Processes with project path" -Status 'BLOCK' -Detail ("$($cwdProcs.Count) processes")
    } else {
        Write-Check -Name "Processes with project path" -Status 'OK' -Detail 'None found via CIM'
    }
} catch {
    $warnings.Add("CIM query failed: $($_.Exception.Message)")
    Write-Check -Name "Processes with project path" -Status 'WARN' -Detail "CIM query failed: $($_.Exception.Message)"
}

# Check 5: Handle.exe (optional dependency)
$handleExe = Get-Command -Name 'handle.exe' -ErrorAction SilentlyContinue
if ($handleExe) {
    try {
        $handleOutput = & handle.exe -a -u $ProjectPath -nobanner 2>&1
        if ($LASTEXITCODE -eq 0 -and $handleOutput -match 'pid:\s*(\d+)') {
            $blockingPid = $matches[1]
            $blockers.Add("Handle.exe found file lock: PID $blockingPid")
            Write-Check -Name "Handle.exe file locks" -Status 'BLOCK' -Detail "Blocking PID: $blockingPid"
        } else {
            Write-Check -Name "Handle.exe file locks" -Status 'OK' -Detail 'No locks detected'
        }
    } catch {
        $warnings.Add("Handle.exe invocation failed: $($_.Exception.Message)")
        Write-Check -Name "Handle.exe file locks" -Status 'WARN' -Detail "Invocation failed"
    }
} else {
    $warnings.Add("Handle.exe not installed (optional — install from Sysinternals for thorough checks)")
    Write-Check -Name "Handle.exe" -Status 'WARN' -Detail 'Not installed (fallback: CIM only)'
}

# Check 6: Go build/test processes
$goProcs = Get-Process -Name 'go','gopls','dlv','go.exe' -ErrorAction SilentlyContinue
if ($goProcs) {
    foreach ($p in $goProcs) {
        $warnings.Add("Go process running: $($p.ProcessName) PID $($p.Id)")
    }
    Write-Check -Name "Go processes (build/test/dlv/gopls)" -Status 'WARN' -Detail "$($goProcs.Count) processes"
} else {
    Write-Check -Name "Go processes (build/test/dlv/gopls)" -Status 'OK' -Detail 'None running'
}

# Check 7: Memory dir exists (informational)
$memDir = "$env:USERPROFILE\.claude\projects\C--Users-ktown-Projects-cc-discord-presence"
if (Test-Path $memDir) {
    Write-Check -Name "Memory dir (old path)" -Status 'OK' -Detail "Exists at $memDir — will be migrated in handoff"
} else {
    $warnings.Add("Memory dir not found at $memDir — unusual state, investigate")
    Write-Check -Name "Memory dir (old path)" -Status 'WARN' -Detail "Not found — unusual"
}

# Summary
Write-Host ""
Write-Host "=========================================="
Write-Host "Summary:"
Write-Host "  Blockers: $($blockers.Count)" -ForegroundColor ($(if ($blockers.Count -gt 0) { 'Red' } else { 'Green' }))
Write-Host "  Warnings: $($warnings.Count)" -ForegroundColor ($(if ($warnings.Count -gt 0) { 'Yellow' } else { 'Green' }))

if ($blockers.Count -gt 0) {
    Write-Host ""
    Write-Host "BLOCKERS (must resolve before handoff):" -ForegroundColor Red
    foreach ($b in $blockers) { Write-Host "  - $b" -ForegroundColor Red }
}

if ($warnings.Count -gt 0) {
    Write-Host ""
    Write-Host "WARNINGS (investigate but may proceed):" -ForegroundColor Yellow
    foreach ($w in $warnings) { Write-Host "  - $w" -ForegroundColor Yellow }
}

if ($DryRun) {
    Write-Host ""
    Write-Host "[DryRun] Exit code forced to 0 regardless of results." -ForegroundColor Cyan
    exit 0
}

if ($blockers.Count -gt 0) {
    Write-Host ""
    Write-Host "Exit 2: BLOCKERS present. DO NOT run handoff.ps1 yet." -ForegroundColor Red
    exit 2
}

if ($warnings.Count -gt 0) {
    Write-Host ""
    Write-Host "Exit 1: Warnings present. Review before proceeding." -ForegroundColor Yellow
    exit 1
}

Write-Host ""
Write-Host "Exit 0: All checks passed. Safe to run handoff.ps1." -ForegroundColor Green
exit 0
```

**Critical Implementation Notes:**
- Use `[CmdletBinding()]` to enable `-Verbose` automatic parameter
- Use `System.Collections.Generic.List[string]` for blockers/warnings (add is O(1))
- All `Get-Process` calls use `-ErrorAction SilentlyContinue` per RESEARCH §2.1
- CIM query uses `Get-CimInstance` (NOT deprecated `Get-WmiObject`)
- `Get-Command handle.exe` checks for Sysinternals tool optionally
- Exit codes: 0=ok, 1=warn, 2=block (matches Plan 03 contract in RESEARCH §7.1)
- `$PID` is built-in PowerShell variable for current shell PID (excluded from CIM results)
