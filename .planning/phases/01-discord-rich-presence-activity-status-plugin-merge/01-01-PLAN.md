package logger

import (

    "log/slog"
    "os"
    "gopkg.in/natefinch/lumberjack.v2"
)

var defaultLogger *slog.Logger

// Setup initializes the global slog logger with file rotation and optional stderr output.
// logFile: path to log file (default ~/.claude/discord-presence.log)
// level: "debug", "info", "error"
// Per D-52: 5MB max file size, rotate on overflow
func Setup(logFile string, level string) {
    lj := &lumberjack.Logger{
        Filename:   logFile,
        MaxSize:    5, // MB per D-52
        MaxBackups: 2,
        MaxAge:     7, // days
        Compress:   false,
    }

    var slogLevel slog.Level
    switch level {
    case "debug":
        slogLevel = slog.LevelDebug
    case "error":
        slogLevel = slog.LevelError
    default:
        slogLevel = slog.LevelInfo
    }

    // Write to both file and stderr
    writer := io.MultiWriter(lj, os.Stderr)
    handler := slog.NewTextHandler(writer, &slog.HandlerOptions{
        Level: slogLevel,
    })
    defaultLogger = slog.New(handler)
    slog.SetDefault(defaultLogger)
}
