   func TestPresetMessageCounts(t *testing.T) {
       presets, err := LoadAll()
       if err != nil {
           t.Fatalf("LoadAll failed: %v", err)
       }

       if len(presets) != 8 {
           t.Fatalf("expected 8 presets, got %d", len(presets))
       }

       minMessages := 40
       icons := AllActivityIcons()

       for name, p := range presets {
           for _, icon := range icons {
               pool := p.SingleSessionDetails[icon]
               if len(pool) < minMessages {
                   t.Errorf("%s: singleSessionDetails[%s] has %d messages, want >= %d", name, icon, len(pool), minMessages)
               }
           }
           if len(p.SingleSessionState) < minMessages {
               t.Errorf("%s: singleSessionState has %d messages, want >= %d", name, len(p.SingleSessionState), minMessages)

           if len(p.SingleSessionDetailsFallback) < 15 {
               t.Errorf("%s: singleSessionDetailsFallback has %d messages, want >= 15", name, len(p.SingleSessionDetailsFallback))
           }
       }
   }

   func TestPresetNewPlaceholders(t *testing.T) {
       presets, err := LoadAll()
       if err != nil {
           t.Fatalf("LoadAll failed: %v", err)
       }

       newPlaceholders := []string{"{file}", "{command}", "{query}"}

       for name, p := range presets {
           codingPool := p.SingleSessionDetails["coding"]
           hasNew := 0
           for _, msg := range codingPool {
               for _, ph := range newPlaceholders {
                   if strings.Contains(msg, ph) {
                       hasNew++
                       break
                   }
               }
           }
           // At least 30% should use new placeholders (12 out of 40)
           if hasNew < 12 {
               t.Errorf("%s coding: only %d/%d messages use new placeholders ({file}/{command}/{query}), want >= 12", name, hasNew, len(codingPool))
           }
       }
   }
