Create `.goreleaser.yaml` at repo root with this exact content:

```yaml
# yaml-language-server: $schema=https://goreleaser.com/static/schema.json
version: 2

project_name: dsrcode

builds:
  - id: dsrcode
    main: .
    binary: dsrcode
    env:
      - CGO_ENABLED=0
    goos:
      - darwin
      - linux
      - windows
    goarch:
      - amd64
      - arm64
    ignore:
      - goos: windows
        goarch: arm64
    mod_timestamp: "{{ .CommitTimestamp }}"

archives:
  - id: default
    formats:
      - tar.gz
    format_overrides:
      - goos: windows
        formats:
          - zip
    name_template: >-
      {{ .ProjectName }}_{{ .Version }}_{{ .Os }}_{{ .Arch }}
    files:
      - LICENSE
      - README.md
      - CHANGELOG.md
      - PRIVACY.md
      - TERMS.md

checksum:
  name_template: "{{ .ProjectName }}_{{ .Version }}_checksums.txt"
  algorithm: sha256

changelog:
  use: github
  groups:
    - title: Features
      regexp: '^.*?feat(\([[:word:]]+\))??!?:.+$'
      order: 0
    - title: Bug Fixes
      regexp: '^.*?fix(\([[:word:]]+\))??!?:.+$'
      order: 1
    - title: Documentation
      regexp: '^.*?docs(\([[:word:]]+\))??!?:.+$'
      order: 2
    - title: Other
      order: 999
  filters:
    exclude:
      - '^test:'
      - '^ci:'
      - '^chore:'

release:
  github:
    owner: StrainReviews
    name: dsrcode
  prerelease: auto
  make_latest: true
```

- No custom ldflags section -- GoReleaser v2 default ldflags inject main.version, main.commit, main.date automatically (per DIST-06/DIST-07, lowercase variables)
- `ignore: windows/arm64` per DIST-38
- `formats: [tar.gz]` with Windows override to `zip` per DIST-11 (v2 syntax: list not scalar)
- `checksum.algorithm: sha256` per DIST-12
- `changelog.use: github` with groups per DIST-13
- `prerelease: auto` and `make_latest: true` per DIST-08
- `release.github.owner: StrainReviews`, `name: dsrcode` per DIST-01
- `files:` includes LICENSE, README.md, CHANGELOG.md, PRIVACY.md, TERMS.md per DIST-11
