The new logic:

```go
func (r *SessionRegistry) StartSession(req ActivityRequest, pid int) *Session {
    r.mu.Lock()
    defer r.mu.Unlock()

    newSource := sourceFromID(req.SessionID)

    // Phase 1 dedup: exact ProjectPath+PID match still returns existing (backward compat)
    for _, existing := range r.sessions {
        if existing.ProjectPath == req.Cwd && existing.PID == pid {
            return existing
        }
    }

    // Phase 3 dedup: source-aware upgrade for same project (D-01, D-02)
    // Only applies when new session is synthetic (PID=0) or matches existing synthetic
    existing := r.foundExisting(filepath.Base(req.Cwd))
    if existing != nil {
        existingRank := sourceRank(existing.Source)
        newRank := sourceRank(newSource)

        if newRank > existingRank {
            // Upgrade: higher-quality source replaces lower (D-04)
            slog.Info("session upgraded",
                "from", existing.SessionID,
                "fromSource", existing.Source.String(),
                "to", req.SessionID,
                "toSource", newSource.String(),
                "project", existing.ProjectName,
            )
            upgraded := r.upgradeSession(existing, req.SessionID, pid, newSource)
            r.notifyChange()
            return upgraded
        }

        if newRank <= existingRank {
            // Skip: same or lower quality source (no downgrade)
            slog.Debug("synthetic session skipped",
                "existing", existing.SessionID,
                "existingSource", existing.Source.String(),
                "attempted", req.SessionID,
                "attemptedSource", newSource.String(),
            )
            return existing
        }
    }

    // No existing match: create new session
    now := time.Now()
    s := &Session{
        SessionID:      req.SessionID,
        ProjectPath:    req.Cwd,
        ProjectName:    filepath.Base(req.Cwd),
        PID:            pid,
        Source:         newSource,
        Details:        "Starting session...",
        SmallImageKey:  "starting",
        SmallText:      "Starting up...",
        Status:         StatusActive,
        ActivityCounts: EmptyActivityCounts(),
        StartedAt:      now,
        LastActivityAt: now,
    }

    r.sessions[req.SessionID] = s
    r.notifyChange()
    return s
}
```

CRITICAL NUANCE for D-01: The foundExisting search finds ANY session with the same ProjectName. But if both sessions have real PIDs (both > 0, both non-synthetic), they are genuinely different Claude Code windows and should BOTH exist. The upgrade logic handles this correctly because:
- Real window 1: SourceClaude (UUID), rank=2
- Real window 2: SourceClaude (UUID), rank=2
- newRank (2) is NOT > existingRank (2), so the `if newRank > existingRank` branch is skipped
- newRank (2) <= existingRank (2), so `return existing` -- BUT this is wrong! We need Window 2 to create its own session.

Fix: After the foundExisting check, add a guard that only triggers the upgrade/skip logic when at least one side is synthetic:

```go
if existing != nil && (existing.Source != SourceClaude || newSource != SourceClaude) {
    // At least one side is synthetic -- apply upgrade/skip logic
    // ...
}
// If both are SourceClaude (real UUIDs), fall through to create new session
```

This ensures two real Claude Code windows on the same project each get their own session (D-01), while synthetic duplicates (JSONL or HTTP alongside a real session) get merged.

```go
func ingestJSONLFallback(registry *session.SessionRegistry) {
    sessionData := readSessionData()
    if sessionData == nil {
        return
    }

    syntheticID := "jsonl-" + sessionData.ProjectName

    jsonlDeprecationOnce.Do(func() {
        slog.Warn("JSONL fallback active -- configure HTTP hooks for better accuracy",
            "project", sessionData.ProjectName,
            "hint", "Run /dsrcode:setup to migrate to HTTP hooks",
        )
    })

    // Ensure the passive session exists (registry handles dedup via D-02)
    existing := registry.GetSession(syntheticID)
    if existing == nil {
        registry.StartSession(session.ActivityRequest{
            SessionID: syntheticID,
            Cwd:       sessionData.ProjectPath,
            Details:   fmt.Sprintf("Working on %s", sessionData.ProjectName),
        }, 0)
    }

    registry.UpdateSessionData(
        syntheticID,
        sessionData.ModelName,
        sessionData.GitBranch,
        sessionData.TotalTokens,
        sessionData.TotalCost,
    )
}
```
