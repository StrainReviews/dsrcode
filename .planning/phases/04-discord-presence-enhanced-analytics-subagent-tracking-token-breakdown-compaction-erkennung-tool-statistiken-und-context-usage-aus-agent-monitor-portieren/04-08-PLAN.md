// UpdateTranscriptPath sets the JSONL transcript file path for a session per D-18.
func (r *SessionRegistry) UpdateTranscriptPath(sessionID, path string) {
    r.mu.Lock()
    defer r.mu.Unlock()
    
    s, ok := r.sessions[sessionID]
    if !ok {
        return
    }
    
    updated := *s // copy-before-modify
