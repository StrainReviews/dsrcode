func (s *Server) handleSessionEnd(w http.ResponseWriter, r *http.Request) {
    // Always return 200 per D-16
    w.WriteHeader(http.StatusOK)

    var payload HookPayload
    if err := json.NewDecoder(r.Body).Decode(&payload); err != nil {
        slog.Debug("session-end: invalid JSON", "error", err)
        return
    }
    if payload.SessionID == "" {
        return
    }

    reason := payload.Reason
    slog.Info("session-end received", "session", payload.SessionID, "reason", reason)

    // Background goroutine: final JSONL read then remove
    go func() {
        if s.tracker != nil && payload.TranscriptPath != "" {
            result, err := analytics.ParseTranscript(payload.TranscriptPath)
            if err == nil {
                for model, tokens := range result.Tokens {
                    s.tracker.UpdateTokens(payload.SessionID, model, tokens)
                }
            }
        }
        s.registry.EndSession(payload.SessionID)
    }()

    s.hookStats.record("session-end")
}
```

func (s *Server) handlePostToolUse(w http.ResponseWriter, r *http.Request) {
    w.WriteHeader(http.StatusOK)

    var payload HookPayload
    if err := json.NewDecoder(r.Body).Decode(&payload); err != nil {
        slog.Debug("post-tool-use: invalid JSON", "error", err)
        return
    }
    if payload.SessionID == "" {
        return
    }

    // Record tool usage
    if s.tracker != nil {
        s.tracker.RecordTool(payload.SessionID, payload.ToolName)
    }

    // Throttled JSONL read: max 1 per 10s per session (D-02)
    if s.tracker != nil && payload.TranscriptPath != "" {
        now := time.Now()
        key := payload.SessionID
        if lastVal, ok := s.lastTranscriptRead.Load(key); ok {
            if now.Sub(lastVal.(time.Time)) < 10*time.Second {
                goto skipRead
            }
        }
        s.lastTranscriptRead.Store(key, now)
        go func() {
            result, err := analytics.ParseTranscript(payload.TranscriptPath)
            if err != nil {
                slog.Debug("post-tool-use: transcript parse failed", "error", err)
                return
            }
            for model, tokens := range result.Tokens {
                s.tracker.UpdateTokens(payload.SessionID, model, tokens)
            }
        }()
    }
skipRead:

    // Clear error state if previously in error (D-19: back after successful PostToolUse)
    existing := s.registry.GetSession(payload.SessionID)
    if existing != nil && existing.SmallImageKey == "error" {
        s.registry.UpdateActivity(payload.SessionID, session.ActivityRequest{
            SessionID:     payload.SessionID,
            SmallImageKey: "coding",
            SmallText:     "Back to work",
        })
    }

    if payload.TranscriptPath != "" {
        s.registry.UpdateTranscriptPath(payload.SessionID, payload.TranscriptPath)
    }

    s.hookStats.record("post-tool-use")
    slog.Debug("post-tool-use received", "session", payload.SessionID, "tool", payload.ToolName)
}
```

// StopFailurePayload extends HookPayload with error-specific fields.
type StopFailurePayload struct {
    SessionID          string `json:"session_id"`
    Cwd                string `json:"cwd"`
    TranscriptPath     string `json:"transcript_path"`
    Error              string `json:"error"`         // rate_limit, authentication_failed, billing_error, etc.
    ErrorDetails       string `json:"error_details"`
    LastAssistantMsg   string `json:"last_assistant_message"`
}

func (s *Server) handleStopFailure(w http.ResponseWriter, r *http.Request) {
    w.WriteHeader(http.StatusOK)

    var payload StopFailurePayload
    if err := json.NewDecoder(r.Body).Decode(&payload); err != nil {
        slog.Debug("stop-failure: invalid JSON", "error", err)
        return
    }
    if payload.SessionID == "" {
        return
    }

    // Map error type to display text
    errorText := "API Error"
    switch payload.Error {
    case "rate_limit":
        errorText = "Rate Limited"
    case "authentication_failed":
        errorText = "Auth Error"
    case "billing_error":
        errorText = "Billing Error"
    case "max_output_tokens":
        errorText = "Max Tokens"
    case "server_error":
        errorText = "Server Error"
    case "invalid_request":
        errorText = "Invalid Request"
    }

    // Set error state on session (D-19: SmallImage=error, SmallText=error type)
    s.registry.UpdateActivity(payload.SessionID, session.ActivityRequest{
        SessionID:     payload.SessionID,
        SmallImageKey: "error",
        SmallText:     errorText,
    })

    slog.Warn("stop-failure received", "session", payload.SessionID, "error", payload.Error, "details", payload.ErrorDetails)
    s.hookStats.record("stop-failure")
}

func (s *Server) handlePreCompact(w http.ResponseWriter, r *http.Request) {

    if newProject == "" || newProject == "." {
        newProject = "Unknown Project"
    }
    s.registry.UpdateActivity(payload.SessionID, session.ActivityRequest{
        SessionID: payload.SessionID,
        Cwd:       payload.NewCwd,
        Details:   fmt.Sprintf("Working on %s", newProject),
    })

    slog.Info("cwd-changed received", "session", payload.SessionID, "from", payload.OldCwd, "to", payload.NewCwd)
    s.hookStats.record("cwd-changed")
}
