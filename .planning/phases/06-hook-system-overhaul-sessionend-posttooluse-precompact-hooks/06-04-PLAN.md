1. Add auto-exit goroutine after registry creation (after line ~210 area, after HTTP server start):

```go
// Auto-exit goroutine: shutdown daemon when all sessions end (D-04)
go func() {
    if cfg.ShutdownGracePeriod == 0 {
        slog.Info("auto-exit disabled (shutdownGracePeriod=0)")
        return
    }

    var graceTimer *time.Timer
    var graceCancel context.CancelFunc

    // Create a channel that fires when registry changes
    // We piggyback on the existing onChange callback
    autoExitChan := make(chan struct{}, 1)

    // Wrap the existing onChange to also signal auto-exit check
    originalOnChange := registry.SwapOnChange(func() {
        // Signal debounce channel (original behavior)
        select {
        case updateChan <- struct{}{}:
        default:
        }
        // Signal auto-exit check
        select {
        case autoExitChan <- struct{}{}:
        default:
        }
    })
    _ = originalOnChange // was already wired to updateChan

    for {
        select {
        case <-ctx.Done():
            if graceTimer != nil {
                graceTimer.Stop()
            }
            return
        case <-autoExitChan:
            count := registry.SessionCount()

            if count == 0 {
                // All sessions gone — start grace period (D-04, D-05)
                if graceTimer != nil {
                    continue // already counting down
                }
                slog.Info("all sessions ended, starting grace period", "duration", cfg.ShutdownGracePeriod)

                var graceCtx context.Context
                graceCtx, graceCancel = context.WithCancel(ctx)

                graceTimer = time.AfterFunc(cfg.ShutdownGracePeriod, func() {
                    select {
                    case <-graceCtx.Done():
                        return // cancelled
                    default:
                    }
                    slog.Info("grace period expired, initiating auto-exit")
                    cancel() // triggers main shutdown
                })
            } else if graceTimer != nil {
                // New session arrived during grace period — abort (D-06)
                slog.Info("new session during grace period, cancelling shutdown", "sessions", count)
                graceTimer.Stop()
                graceTimer = nil
                if graceCancel != nil {
                    graceCancel()
                    graceCancel = nil
                }
            }
        }
    }
}()
```
